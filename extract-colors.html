Intive Brand Colors - Visual Extraction Tool

Open intive.com in DevTools and run this in the console:

// Extract all computed colors from intive.com
const colors = new Set();
document.querySelectorAll('*').forEach(el => {
    const style = window.getComputedStyle(el);
    ['color', 'backgroundColor', 'borderColor'].forEach(prop => {
        const val = style[prop];
        if (val && val !== 'rgba(0, 0, 0, 0)' && val !== 'transparent') {
            colors.add(val);
        }
    });
});

// Convert RGB to HEX
function rgbToHex(rgb) {
    const match = rgb.match(/\d+/g);
    if (!match) return rgb;
    return '#' + match.slice(0,3).map(x => {
        const hex = parseInt(x).toString(16);
        return hex.length === 1 ? '0' + hex : hex;
    }).join('');
}

// Show unique colors
Array.from(colors).map(c => {
    return c.includes('rgb') ? rgbToHex(c) : c;
}).filter(c => c.startsWith('#')).forEach(c => {
    console.log(c);
});

Manual Instructions:

  1. Open https://intive.com in Chrome/Edge
  2. Press F12 to open DevTools
  3. Go to Console tab
  4. Paste the script above
  5. Copy the hex colors that appear
  6. Save them to BRAND_COLORS.md
